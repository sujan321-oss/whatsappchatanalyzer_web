#define MAX_BUFFER_SIZE 1024

int main() {
    int sockfd;
    struct sockaddr_in server_address, client_address;
    char buffer[MAX_BUFFER_SIZE];
    socklen_t client_length;
    FILE *file;

    // Create a UDP socket
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    if (sockfd < 0) {
        perror("Failed to create socket");
        return 1;
    }

    // Bind the socket to a specific address and port
    server_address.sin_family = AF_INET;
    server_address.sin_port = htons(12345);
    server_address.sin_addr.s_addr = INADDR_ANY;
    if (bind(sockfd, (struct sockaddr *)&server_address, sizeof(server_address)) < 0) {
        perror("Binding failed");
        return 1;
    }

    // Receive the file name from the client
    client_length = sizeof(client_address);
    memset(buffer, 0, MAX_BUFFER_SIZE);
    if (recvfrom(sockfd, buffer, MAX_BUFFER_SIZE, 0, (struct sockaddr *)&client_address, &client_length) < 0) {
        perror("Failed to receive file name");
        return 1;
    }
    char *file_name = buffer;

    // Open the file in write mode
    file = fopen(file_name, "wb");
    if (file == NULL) {
        perror("Failed to open file");
        return 1;
    }

    // Receive the file data and write it to the file
    while (1) {
        memset(buffer, 0, MAX_BUFFER_SIZE);
        ssize_t bytes_received = recvfrom(sockfd, buffer, MAX_BUFFER_SIZE, 0, (struct sockaddr *)&client_address, &client_length);
        if (bytes_received < 0) {
            perror("Failed to receive data");
            return 1;
        }
        if (bytes_received == 0) {
            break;  // Transfer complete
        }
        fwrite(buffer, sizeof(char), bytes_received, file);
    }

    // Send acknowledgment to the client
    const char *ack = "File transferred successfully!";
    if (sendto(sockfd, ack, strlen(ack), 0, (struct sockaddr *)&client_address, client_length) < 0) {
        perror("Failed to send acknowledgment");
        return 1;
    }

    // Close the file and socket
    fclose(file);
    close(sockfd);

    return 0;
}
